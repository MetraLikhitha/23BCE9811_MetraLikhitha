Test 2:

1)List the Employee Name, their Department Name, and the Location Name for all employees.

Query:
SELECT 
    e.employee_name AS Employee_Name,
    d.dept_name AS Department_Name,
    d.location_id AS Location_Name
FROM Employees e
JOIN Departments d
    ON e.dept_id = d.dept_id;

2)Retrieve the Project Name, Client Name, and the Total Hours Worked on that project, but only for projects managed by the 'Consulting' department.

Query:
SELECT 
    p.project_name AS Project_Name,
    p.client_name AS Client_Name,
    SUM(a.hours_worked) AS Total_Hours_Worked
FROM Projects p
JOIN Departments d
    ON p.dept_id = d.dept_id
JOIN Assignments a
    ON p.project_id = a.project_id
WHERE d.dept_name = 'Consulting'
GROUP BY p.project_id, p.project_name, p.client_name;

3)Find the names of Employees who have worked on a project with Client Name 'GlobalTech Solutions'. List the Employee Name and Project Name

Query:
SELECT 
    e.employee_name AS Employee_Name,
    p.project_name AS Project_Name
FROM Employees e
JOIN Assignments a
    ON e.employee_id = a.employee_id
JOIN Projects p
    ON a.project_id = p.project_id
WHERE p.client_name = 'GlobalTech Solutions';

4)List all Departments that have no Employees currently assigned to them.

Query:
SELECT 
    d.dept_name AS Department_Name
FROM Departments d
LEFT JOIN Employees e
    ON d.dept_id = e.dept_id
WHERE e.employee_id IS NULL;

5)Calculate the Average Hours Worked per Employee across all projects. List the Employee Name and the average hours

Query:
SELECT 
    e.employee_name AS Employee_Name,
    AVG(a.hours_worked) AS Average_Hours_Worked
FROM Employees e
JOIN Assignments a
    ON e.employee_id = a.employee_id
GROUP BY e.employee_id, e.employee_name;

6)Calculate the total number of employees assigned to each Project, but only include projects that have more than 5 employees assigned.

Query:
SELECT 
    p.project_name AS Project_Name,
    COUNT(a.employee_id) AS Total_Employees
FROM Projects p
JOIN Assignments a
    ON p.project_id = a.project_id
GROUP BY p.project_id, p.project_name
HAVING COUNT(a.employee_id) > 5;

7)Retrieve the Department Name(s) that have employees assigned to more than one project.

Query:
SELECT 
    d.dept_name AS Department_Name
FROM Departments d
JOIN Employees e
    ON d.dept_id = e.dept_id
JOIN Assignments a
    ON e.employee_id = a.employee_id
GROUP BY d.dept_id, d.dept_name
HAVING COUNT(DISTINCT a.project_id) > 1;

8)Write a query to list the Employee Name and Project Name for any employee who has worked less than 10 hours on a project.

Query:
SELECT 
    e.employee_name AS Employee_Name,
    p.project_name AS Project_Name
FROM Employees e
JOIN Assignments a
    ON e.employee_id = a.employee_id
JOIN Projects p
    ON a.project_id = p.project_id
WHERE a.hours_worked < 10;

9)Find the Project Name(s) that have no assigned employees.

Query:
SELECT 
    p.project_name AS Project_Name
FROM Projects p
LEFT JOIN Assignments a
    ON p.project_id = a.project_id
WHERE a.employee_id IS NULL;

10)Write a MySQL query using the DATE_FORMAT function to list all projects that were active in the year 2023.

Query:
SELECT 
    project_name,
    client_name,
    start_date
FROM Projects
WHERE DATE_FORMAT(start_date, '%D') = '2023';

11)Find the top 3 Departments based on the total hours worked by all their employees across all projects.

Query:
SELECT 
    d.dept_name AS Department_Name,
    SUM(a.hours_worked) AS Total_Hours_Worked
FROM Departments d
JOIN Employees e
    ON d.dept_id = e.dept_id
JOIN Assignments a
    ON e.employee_id = a.employee_id
GROUP BY d.dept_id, d.dept_name
ORDER BY Total_Hours_Worked DESC
LIMIT 3;

12)Find the Employee Name(s) who are assigned to the same projects as Employee ID 101

Query:
SELECT DISTINCT 
    e.employee_name AS Employee_Name
FROM Employees e
JOIN Assignments a
    ON e.employee_id = a.employee_id
WHERE a.project_id IN (
    SELECT project_id
    FROM Assignments
    WHERE employee_id = 101
)
AND e.employee_id <> 101;

13)Select the Employee Name(s) whose salary is greater than the maximum salary in their own department.

Query:
SELECT 
    e.employee_name AS Employee_Name
FROM Employees e
WHERE e.salary > (
    SELECT MAX(e2.salary)
    FROM Employees e2
    WHERE e2.dept_id = e.dept_id
      AND e2.employee_id <> e.employee_id
);

14)Show the Manager Name, the Department Name they manage, and the total number of direct reports they have

Query:
SELECT 
    m.employee_name AS Manager_Name,
    d.dept_name AS Department_Name,
    COUNT(e.employee_id) AS Total_Direct_Reports
FROM Employees e
JOIN Employees m
    ON e.manager_id = m.employee_id
JOIN Departments d
    ON m.dept_id = d.dept_id
GROUP BY m.employee_id, m.employee_name, d.dept_name;

15)Find the Department Name(s) that have an average salary above the company-wide average salary.

Query:
SELECT 
    d.dept_name AS Department_Name,
    AVG(e.salary) AS Average_Department_Salary
FROM Departments d
JOIN Employees e
    ON d.dept_id = e.dept_id
GROUP BY d.dept_id, d.dept_name
HAVING AVG(e.salary) > (
    SELECT AVG(salary)
    FROM Employees
);

16)Identify the Client Name(s) whose projects have a total combined hours worked exceeding 500 hours.

Query:
SELECT 
    p.client_name AS Client_Name,
    SUM(a.hours_worked) AS Total_Hours_Worked
FROM Projects p
JOIN Assignments a
    ON p.project_id = a.project_id
GROUP BY p.client_name
HAVING SUM(a.hours_worked) > 500;
